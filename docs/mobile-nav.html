[aria-expanded="false"] {
  display: none;
}

/* Position expand button at corner of parent */
.expand-parent {
  position: relative;
}

  position: absolute;
  right: 0;
  top: 0;
}

/* Flip the icon when active */
.expand-btn i {
  transition: transform 0.3s ease;
}
.expand-btn.active i {
  transform: rotate(180deg);
}
