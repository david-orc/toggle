/* Hide collapsed */
[aria-expanded="false"] {
  display: none;
}

/* Horizontal menu */

}

/* Dropdown menus */
.toggle-parent {
  position: relative;
}
.toggle-parent > ul {
  left: 0;
  position: absolute;
  top: 100%;
}

/* Deep dropdown menus (pop-out right) */
.toggle-parent .toggle-parent > ul {
  left: 100%;
  top: 0;
}

/* Add icon to parents */
.toggle-parent > a::after {
  content: '\f107'; /* Font Awesome 'angle-down' */
  display: inline-block;
  font-family: 'FontAwesome';
  margin-left: 0.5rem;
  transition: transform 0.3s ease;
}

/* Add different icon to deep parents */
.toggle-parent .toggle-parent > a::after {
  content: '\f105'; /* Font Awesome 'angle-right' */
}

/* Flip the icon */
.toggle-parent > a.toggle-btn-active::after {
  transform: rotate(180deg);
